وَٱلصَّٰٓفَّٰتِ صَفّٗا _ فَٱلزَّٰجِرَٰتِ زَجۡرٗا _ فَٱلتَّٰلِيَٰتِ ذِكۡرًا _ إِنَّ إِلَٰهَكُمُۥ لَوَٰحِدٞ _ رَّبُّ ٱلسَّمَٰوَٰتِ وَٱلۡأَرۡضِ وَمَا بَيۡنَهُمَا وَرَبُّ ٱلۡمَشَٰرِقِ _ إِنَّا زَيَّنَّا ٱلسَّمَآءَ ٱلدُّنۡيَا بِزِينَةِ ٱلۡكَوَاكِبِ _ وَحِفۡظٗا مِّن كُلِّ شَيۡطَٰنٖ مَّارِدٖ _ لَّا يَسۡمَعُونَ إِلَى ٱلۡمَلَإِ ٱلۡأَعۡلَىٰ وَيُقۡذَفُونَ مِن كُلِّ جَانِبٖ _ دُحُورٗاۖ وَلَهُمُۥ عَذَابٞ وَاصِبٌ _ إِلَّا مَنۡ خَطِفَ ٱلۡخَطۡفَةَ فَأَتۡبَعَهُۥ شِهَابٞ ثَاقِبٞ _ فَٱسۡتَفۡتِهِمُۥ أَهُمُۥ أَشَدُّ خَلۡقًا أَم مَّنۡ خَلَقۡنَاۚ إِنَّا خَلَقۡنَٰهُمُۥ مِن طِينٖ لَّازِبِۭ _ بَلۡ عَجِبۡتَ وَيَسۡخَرُونَ _ وَإِذَا ذُكِّرُواْ لَا يَذۡكُرُونَ _ وَإِذَا رَأَوۡاْ ءَايَةٗ يَسۡتَسۡخِرُونَ _ وَقَالُواْ إِنۡ هَٰذَا إِلَّا سِحۡرٞ مُّبِينٌ _ أَ۟ذَا مُتۡنَا وَكُنَّا تُرَابٗا وَعِظَٰمًا أَ۟نَّا لَمَبۡعُوثُونَ _ أَوَ ءَابَآؤُنَا ٱلۡأَوَّلُونَ _ قُلۡ نَعَمۡ وَأَنتُمُۥ دَٰخِرُونَ _ فَإِنَّمَا هِيَ زَجۡرَةٞ وَٰحِدَةٞ فَإِذَا هُمُۥ يَنظُرُونَ _ وَقَالُواْ يَٰوَيۡلَنَا هَٰذَا يَوۡمُ ٱلدِّينِ _ هَٰذَا يَوۡمُ ٱلۡفَصۡلِ ٱلَّذِي كُنتُمُۥ بِهِۦ تُكَذِّبُونَ _ ۞ٱحۡشُرُواْ ٱلَّذِينَ ظَلَمُواْ وَأَزۡوَٰجَهُمُۥ وَمَا كَانُواْ يَعۡبُدُونَ _ مِن دُونِ ٱللَّهِ فَٱهۡدُوهُمُۥ إِلَىٰ صِرَٰطِ ٱلۡجَحِيمِ _ وَقِفُوهُمُۥۖ إِنَّهُمُۥ مَسۡـُٔولُونَ _ مَا لَكُمُۥ لَآ تَّنَاصَرُونَ _ بَلۡ هُمُ ٱلۡيَوۡمَ مُسۡتَسۡلِمُونَ _ وَأَقۡبَلَ بَعۡضُهُمُۥ عَلَىٰ بَعۡضٖ يَتَسَآءَلُونَ _ قَالُواْ إِنَّكُمُۥ كُنتُمُۥ تَأۡتُونَنَا عَنِ ٱلۡيَمِينِ _ قَالُواْ بَل لَّمۡ تَكُونُواْ مُؤۡمِنِينَ _ وَمَا كَانَ لَنَا عَلَيۡكُمُۥ مِن سُلۡطَٰنِۭۖ بَلۡ كُنتُمُۥ قَوۡمٗا طَٰغِينَ _ فَحَقَّ عَلَيۡنَا قَوۡلُ رَبِّنَاۖ إِنَّا لَذَآئِقُونَ _ فَأَغۡوَيۡنَٰكُمُۥ إِنَّا كُنَّا غَٰوِينَ _ فَإِنَّهُمُۥ يَوۡمَئِذٖ فِي ٱلۡعَذَابِ مُشۡتَرِكُونَ _ إِنَّا كَذَٰلِكَ نَفۡعَلُ بِٱلۡمُجۡرِمِينَ _ إِنَّهُمُۥ كَانُواْ إِذَا قِيلَ لَهُمُۥ لَا إِلَٰهَ إِلَّا ٱللَّهُ يَسۡتَكۡبِرُونَ _ وَيَقُولُونَ أَى۪نَّا لَتَارِكُواْ ءَالِهَتِنَا لِشَاعِرٖ مَّجۡنُونِۭ _ بَلۡ جَآءَ بِٱلۡحَقِّ وَصَدَّقَ ٱلۡمُرۡسَلِينَ _ إِنَّكُمُۥ لَذَآئِقُواْ ٱلۡعَذَابِ ٱلۡأَلِيمِ _ وَمَا تُجۡزَوۡنَ إِلَّا مَا كُنتُمُۥ تَعۡمَلُونَ _ إِلَّا عِبَادَ ٱللَّهِ ٱلۡمُخۡلِصِينَ _ أُوْلَٰٓئِكَ لَهُمُۥ رِزۡقٞ مَّعۡلُومٞ _ فَوَٰكِهُ وَهُمُۥ مُكۡرَمُونَ _ فِي جَنَّٰتِ ٱلنَّعِيمِ _ عَلَىٰ سُرُرٖ مُّتَقَٰبِلِينَ _ يُطَافُ عَلَيۡهِمُۥ بِكَأۡسٖ مِّن مَّعِينِۭ _ بَيۡضَآءَ لَذَّةٖ لِّلشَّٰرِبِينَ _ لَا فِيهَا غَوۡلٞ وَلَا هُمُۥ عَنۡهَا يُنزَفُونَ _ وَعِندَهُمُۥ قَٰصِرَٰتُ ٱلطَّرۡفِ عِينٞ _ كَأَنَّهُنَّ بَيۡضٞ مَّكۡنُونٞ _ فَأَقۡبَلَ بَعۡضُهُمُۥ عَلَىٰ بَعۡضٖ يَتَسَآءَلُونَ _ قَالَ قَآئِلٞ مِّنۡهُمُۥ إِنِّي كَانَ لِي قَرِينٞ _ يَقُولُ أَ۟نَّكَ لَمِنَ ٱلۡمُصَدِّقِينَ _ أَ۟ذَا مُتۡنَا وَكُنَّا تُرَابٗا وَعِظَٰمًا أَ۟نَّا لَمَدِينُونَ _ قَالَ هَلۡ أَنتُمُۥ مُطَّلِعُونَ _ فَٱطَّلَعَ فَرَءَاهُۥ فِي سَوَآءِ ٱلۡجَحِيمِ _ قَالَ تَٱللَّهِ إِن كِدتَّ لَتُرۡدِينِ _ وَلَوۡلَا نِعۡمَةُ رَبِّي لَكُنتُ مِنَ ٱلۡمُحۡضَرِينَ _ أَفَمَا نَحۡنُ بِمَيِّتِينَ _ إِلَّا مَوۡتَتَنَا ٱلۡأُولَىٰ وَمَا نَحۡنُ بِمُعَذَّبِينَ _ إِنَّ هَٰذَا لَهُوَ ٱلۡفَوۡزُ ٱلۡعَظِيمُ _ لِمِثۡلِ هَٰذَا فَلۡيَعۡمَلِ ٱلۡعَٰمِلُونَ _ أَذَٰلِكَ خَيۡرٞ نُّزُلًا أَمۡ شَجَرَةُ ٱلزَّقُّومِ _ إِنَّا جَعَلۡنَٰهَا فِتۡنَةٗ لِّلظَّٰلِمِينَ _ إِنَّهَا شَجَرَةٞ تَخۡرُجُ فِي أَصۡلِ ٱلۡجَحِيمِ _ طَلۡعُهَا كَأَنَّهُۥ رُءُوسُ ٱلشَّيَٰطِينِ _ فَإِنَّهُمُۥ لَأٓكِلُونَ مِنۡهَا فَمَالِـُٔونَ مِنۡهَا ٱلۡبُطُونَ _ ثُمَّ إِنَّ لَهُمُۥ عَلَيۡهَا لَشَوۡبٗا مِّنۡ حَمِيمٖ _ ثُمَّ إِنَّ مَرۡجِعَهُمُۥ لَإِلَى ٱلۡجَحِيمِ _ إِنَّهُمُۥ أَلۡفَوۡاْ ءَابَآءَهُمُۥ ضَآلِّينَ _ فَهُمُۥ عَلَىٰ ءَاثَٰرِهِمُۥ يُهۡرَعُونَ _ وَلَقَدۡ ضَلَّ قَبۡلَهُمُۥ أَكۡثَرُ ٱلۡأَوَّلِينَ _ وَلَقَدۡ أَرۡسَلۡنَا فِيهِمُۥ مُنذِرِينَ _ فَٱنظُرۡ كَيۡفَ كَانَ عَٰقِبَةُ ٱلۡمُنذَرِينَ _ إِلَّا عِبَادَ ٱللَّهِ ٱلۡمُخۡلِصِينَ _ وَلَقَدۡ نَادَىٰنَا نُوحٞ فَلَنِعۡمَ ٱلۡمُجِيبُونَ _ وَنَجَّيۡنَٰهُۥ وَأَهۡلَهُۥ مِنَ ٱلۡكَرۡبِ ٱلۡعَظِيمِ _ وَجَعَلۡنَا ذُرِّيَّتَهُۥ هُمُ ٱلۡبَاقِينَ _ وَتَرَكۡنَا عَلَيۡهِۦ فِي ٱلۡأٓخِرِينَ _ سَلَٰمٌ عَلَىٰ نُوحٖ فِي ٱلۡعَٰلَمِينَ _ إِنَّا كَذَٰلِكَ نَجۡزِي ٱلۡمُحۡسِنِينَ _ إِنَّهُۥ مِنۡ عِبَادِنَا ٱلۡمُؤۡمِنِينَ _ ثُمَّ أَغۡرَقۡنَا ٱلۡأٓخَرِينَ _ ۞وَإِنَّ مِن شِيعَتِهِۦ لَإِبۡرَٰهِيمَ _ إِذۡ جَآءَ رَبَّهُۥ بِقَلۡبٖ سَلِيمٍ _ إِذۡ قَالَ لِأَبِيهِۦ وَقَوۡمِهِۦ مَاذَا تَعۡبُدُونَ _ أَى۪فۡكًا ءَالِهَةٗ دُونَ ٱللَّهِ تُرِيدُونَ _ فَمَا ظَنُّكُمُۥ بِرَبِّ ٱلۡعَٰلَمِينَ _ فَنَظَرَ نَظۡرَةٗ فِي ٱلنُّجُومِ _ فَقَالَ إِنِّي سَقِيمٞ _ فَتَوَلَّوۡاْ عَنۡهُۥ مُدۡبِرِينَ _ فَرَاغَ إِلَىٰ ءَالِهَتِهِمُۥ فَقَالَ أَلَا تَأۡكُلُونَ _ مَا لَكُمُۥ لَا تَنطِقُونَ _ فَرَاغَ عَلَيۡهِمُۥ ضَرۡبَۢا بِٱلۡيَمِينِ _ فَأَقۡبَلُواْ إِلَيۡهِۦ يَزِفُّونَ _ قَالَ أَتَعۡبُدُونَ مَا تَنۡحِتُونَ _ وَٱللَّهُ خَلَقَكُمُۥ وَمَا تَعۡمَلُونَ _ قَالُواْ ٱبۡنُواْ لَهُۥ بُنۡيَٰنٗا فَأَلۡقُوهُۥ فِي ٱلۡجَحِيمِ _ فَأَرَادُواْ بِهِۦ كَيۡدٗا فَجَعَلۡنَٰهُمُ ٱلۡأَسۡفَلِينَ _ وَقَالَ إِنِّي ذَاهِبٌ إِلَىٰ رَبِّي سَيَهۡدِينِ _ رَبِّ هَبۡ لِي مِنَ ٱلصَّٰلِحِينَ _ فَبَشَّرۡنَٰهُۥ بِغُلَٰمٍ حَلِيمٖ _ فَلَمَّا بَلَغَ مَعَهُ ٱلسَّعۡيَ قَالَ يَٰبُنَيِّ إِنِّيَ أَرَىٰ فِي ٱلۡمَنَامِ أَنِّيَ أَذۡبَحُكَ فَٱنظُرۡ مَاذَا تَرَىٰۚ قَالَ يَٰأَبَتِ ٱفۡعَلۡ مَا تُؤۡمَرُۖ سَتَجِدُنِي إِن شَآءَ ٱللَّهُ مِنَ ٱلصَّٰبِرِينَ _ فَلَمَّا أَسۡلَمَا وَتَلَّهُۥ لِلۡجَبِينِ _ وَنَٰدَيۡنَٰهُۥ أَن يَٰإِبۡرَٰهِيمُ _ قَدۡ صَدَّقۡتَ ٱلرُّءۡيَاۚ إِنَّا كَذَٰلِكَ نَجۡزِي ٱلۡمُحۡسِنِينَ _ إِنَّ هَٰذَا لَهُوَ ٱلۡبَلَٰٓؤُاْ ٱلۡمُبِينُ _ وَفَدَيۡنَٰهُۥ بِذِبۡحٍ عَظِيمٖ _ وَتَرَكۡنَا عَلَيۡهِۦ فِي ٱلۡأٓخِرِينَ _ سَلَٰمٌ عَلَىٰ إِبۡرَٰهِيمَ _ كَذَٰلِكَ نَجۡزِي ٱلۡمُحۡسِنِينَ _ إِنَّهُۥ مِنۡ عِبَادِنَا ٱلۡمُؤۡمِنِينَ _ وَبَشَّرۡنَٰهُۥ بِإِسۡحَٰقَ نَبِيّٗا مِّنَ ٱلصَّٰلِحِينَ _ وَبَٰرَكۡنَا عَلَيۡهِۦ وَعَلَىٰ إِسۡحَٰقَۚ وَمِن ذُرِّيَّتِهِمَا مُحۡسِنٞ وَظَالِمٞ لِّنَفۡسِهِۦ مُبِينٞ _ وَلَقَدۡ مَنَنَّا عَلَىٰ مُوسَىٰ وَهَٰرُونَ _ وَنَجَّيۡنَٰهُمَا وَقَوۡمَهُمَا مِنَ ٱلۡكَرۡبِ ٱلۡعَظِيمِ _ وَنَصَرۡنَٰهُمُۥ فَكَانُواْ هُمُ ٱلۡغَٰلِبِينَ _ وَءَاتَيۡنَٰهُمَا ٱلۡكِتَٰبَ ٱلۡمُسۡتَبِينَ _ وَهَدَيۡنَٰهُمَا ٱلصِّرَٰطَ ٱلۡمُسۡتَقِيمَ _ وَتَرَكۡنَا عَلَيۡهِمَا فِي ٱلۡأٓخِرِينَ _ سَلَٰمٌ عَلَىٰ مُوسَىٰ وَهَٰرُونَ _ إِنَّا كَذَٰلِكَ نَجۡزِي ٱلۡمُحۡسِنِينَ _ إِنَّهُمَا مِنۡ عِبَادِنَا ٱلۡمُؤۡمِنِينَ _ وَإِنَّ إِلۡيَاسَ لَمِنَ ٱلۡمُرۡسَلِينَ _ إِذۡ قَالَ لِقَوۡمِهِۦ أَلَا تَتَّقُونَ _ أَتَدۡعُونَ بَعۡلٗا وَتَذَرُونَ أَحۡسَنَ ٱلۡخَٰلِقِينَ _ ٱللَّهُ رَبُّكُمُۥ وَرَبُّ ءَابَآئِكُمُ ٱلۡأَوَّلِينَ _ فَكَذَّبُوهُۥ فَإِنَّهُمُۥ لَمُحۡضَرُونَ _ إِلَّا عِبَادَ ٱللَّهِ ٱلۡمُخۡلِصِينَ _ وَتَرَكۡنَا عَلَيۡهِۦ فِي أَجۡمَعِينَ _ إِلَّا عَجُوزٗا فِي ٱلۡغَٰبِرِينَ _ ثُمَّ دَمَّرۡنَا ٱلۡأٓخَرِينَ _ وَإِنَّكُمُۥ لَتَمُرُّونَ عَلَيۡهِمُۥ مُصۡبِحِينَ _ وَبِٱلَّيۡلِۚ أَفَلَا تَعۡقِلُونَ _ وَإِنَّ يُونُسَ لَمِنَ ٱلۡمُرۡسَلِينَ _ إِذۡ أَبَقَ إِلَى ٱلۡفُلۡكِ ٱلۡمَشۡحُونِ _ فَسَاهَمَ فَكَانَ مِنَ ٱلۡمُدۡحَضِينَ _ فَٱلۡتَقَمَهُ ٱلۡحُوتُ وَهُوَ مُلِيمٞ _ فَلَوۡلَا أَنَّهُۥ كَانَ مِنَ ٱلۡمُسَبِّحِينَ _ لَلَبِثَ فِي بَطۡنِهِۦ إِلَىٰ يَوۡمِ يُبۡعَثُونَ _ ۞فَنَبَذۡنَٰهُۥ بِٱلۡعَرَآءِ وَهُوَ سَقِيمٞ _ وَأَنۢبَتۡنَا عَلَيۡهِۦ شَجَرَةٗ مِّن يَقۡطِينٖ _ وَأَرۡسَلۡنَٰهُۥ إِلَىٰ مِاْئَةِ أَلۡفٍ أَوۡ يَزِيدُونَ _ فَـَٔامَنُواْ فَمَتَّعۡنَٰهُمُۥ إِلَىٰ حِينٖ _ فَٱسۡتَفۡتِهِمُۥ أَلِرَبِّكَ ٱلۡبَنَاتُ وَلَهُمُ ٱلۡبَنُونَ _ أَمۡ خَلَقۡنَا ٱلۡمَلَٰٓئِكَةَ إِنَٰثٗا وَهُمُۥ شَٰهِدُونَ _ أَلَا إِنَّهُمُۥ مِنۡ إِفۡكِهِمُۥ لَيَقُولُونَ _ وَلَدَ ٱللَّهُ وَإِنَّهُمُۥ لَكَٰذِبُونَ _ أَصۡطَفَى ٱلۡبَنَاتِ عَلَى ٱلۡبَنِينَ _ مَا لَكُمُۥ كَيۡفَ تَحۡكُمُونَ _ أَفَلَا تَذَّكَّرُونَ _ أَمۡ لَكُمُۥ سُلۡطَٰنٞ مُّبِينٞ _ فَأۡتُواْ بِكِتَٰبِكُمُۥ إِن كُنتُمُۥ صَٰدِقِينَ _ وَجَعَلُواْ بَيۡنَهُۥ وَبَيۡنَ ٱلۡجِنَّةِ نَسَبٗاۚ وَلَقَدۡ عَلِمَتِ ٱلۡجِنَّةُ إِنَّهُمُۥ لَمُحۡضَرُونَ _ سُبۡحَٰنَ ٱللَّهِ عَمَّا يَصِفُونَ _ إِلَّا عِبَادَ ٱللَّهِ ٱلۡمُخۡلِصِينَ _ فَإِنَّكُمُۥ وَمَا تَعۡبُدُونَ _ مَا أَنتُمُۥ عَلَيۡهِۦ بِفَٰتِنِينَ _ إِلَّا مَنۡ هُوَ صَالِ ٱلۡجَحِيمِ _ وَمَا مِنَّا إِلَّا لَهُۥ مَقَامٞ مَّعۡلُومٞ _ وَإِنَّا لَنَحۡنُ ٱلصَّآفُّونَ _ وَإِنَّا لَنَحۡنُ ٱلۡمُسَبِّحُونَ _ وَإِن كَانُواْ لَيَقُولُونَ _ لَوۡ أَنَّ عِندَنَا ذِكۡرٗا مِّنَ ٱلۡأَوَّلِينَ _ لَكُنَّا عِبَادَ ٱللَّهِ ٱلۡمُخۡلِصِينَ _ فَكَفَرُواْ بِهِۦۖ فَسَوۡفَ يَعۡلَمُونَ _ وَلَقَدۡ سَبَقَتۡ كَلِمَتُنَا لِعِبَادِنَا ٱلۡمُرۡسَلِينَ _ إِنَّهُمُۥ لَهُمُ ٱلۡمَنصُورُونَ _ وَإِنَّ جُندَنَا لَهُمُ ٱلۡغَٰلِبُونَ _ فَتَوَلَّ عَنۡهُمُۥ حَتَّىٰ حِينٖ _ وَأَبۡصِرۡهُمُۥ فَسَوۡفَ يُبۡصِرُونَ _ أَفَبِعَذَابِنَا يَسۡتَعۡجِلُونَ _ فَإِذَا نَزَلَ بِسَاحَتِهِمُۥ فَسَآءَ صَبَاحُ ٱلۡمُنذَرِينَ _ وَتَوَلَّ عَنۡهُمُۥ حَتَّىٰ حِينٖ _ وَأَبۡصِرۡ فَسَوۡفَ يُبۡصِرُونَ _ سُبۡحَٰنَ رَبِّكَ رَبِّ ٱلۡعِزَّةِ عَمَّا يَصِفُونَ _ وَسَلَٰمٌ عَلَى ٱلۡمُرۡسَلِينَ _ وَٱلۡحَمۡدُ لِلَّهِ رَبِّ ٱلۡعَٰلَمِينَ _
