الٓمٓۚ أَحَسِبَ ٱلنَّاسُ أَن يُتۡرَكُواْ أَن يَقُولُواْ ءَامَنَّا وَهُمُۥ لَا يُفۡتَنُونَ _ وَلَقَدۡ فَتَنَّا ٱلَّذِينَ مِن قَبۡلِهِمُۥۖ فَلَيَعۡلَمَنَّ ٱللَّهُ ٱلَّذِينَ صَدَقُواْ وَلَيَعۡلَمَنَّ ٱلۡكَٰذِبِينَ _ أَمۡ حَسِبَ ٱلَّذِينَ يَعۡمَلُونَ ٱلسَّيِّـَٔاتِ أَن يَسۡبِقُونَاۚ سَآءَ مَا يَحۡكُمُونَ _ مَن كَانَ يَرۡجُواْ لِقَآءَ ٱللَّهِ فَإِنَّ أَجَلَ ٱللَّهِ لَأٓتٖۚ وَهُوَ ٱلسَّمِيعُ ٱلۡعَلِيمُ _ وَمَن جَٰهَدَ فَإِنَّمَا يُجَٰهِدُ لِنَفۡسِهِۦۚ إِنَّ ٱللَّهَ لَغَنِيٌّ عَنِ ٱلۡعَٰلَمِينَ _ وَٱلَّذِينَ ءَامَنُواْ وَعَمِلُواْ ٱلصَّٰلِحَٰتِ لَنُكَفِّرَنَّ عَنۡهُمُۥ سَيِّـَٔاتِهِمُۥ وَلَنَجۡزِيَنَّهُمُۥ أَحۡسَنَ ٱلَّذِي كَانُواْ يَعۡمَلُونَ _ وَوَصَّيۡنَا ٱلۡإِنسَٰنَ بِوَٰلِدَيۡهِۦ حُسۡنٗاۖ وَإِن جَٰهَدَاكَ لِتُشۡرِكَ بِي مَا لَيۡسَ لَكَ بِهِۦ عِلۡمٞ فَلَا تُطِعۡهُمَاۚ إِلَيَّ مَرۡجِعُكُمُۥ فَأُنَبِّئُكُمُۥ بِمَا كُنتُمُۥ تَعۡمَلُونَ _ وَٱلَّذِينَ ءَامَنُواْ وَعَمِلُواْ ٱلصَّٰلِحَٰتِ لَنُدۡخِلَنَّهُمُۥ فِي ٱلصَّٰلِحِينَ _ وَمِنَ ٱلنَّاسِ مَن يَقُولُ ءَامَنَّا بِٱللَّهِ فَإِذَا أُوذِيَ فِي ٱللَّهِ جَعَلَ فِتۡنَةَ ٱلنَّاسِ كَعَذَابِ ٱللَّهِۖ وَلَئِن جَآءَ نَصۡرٞ مِّن رَّبِّكَ لَيَقُولُنَّ إِنَّا كُنَّا مَعَكُمُۥۚ أَوَلَيۡسَ ٱللَّهُ بِأَعۡلَمَ بِمَا فِي صُدُورِ ٱلۡعَٰلَمِينَ _ وَلَيَعۡلَمَنَّ ٱللَّهُ ٱلَّذِينَ ءَامَنُواْ وَلَيَعۡلَمَنَّ ٱلۡمُنَٰفِقِينَ _ وَقَالَ ٱلَّذِينَ كَفَرُواْ لِلَّذِينَ ءَامَنُواْ ٱتَّبِعُواْ سَبِيلَنَا وَلۡنَحۡمِلۡ خَطَٰيَٰكُمُۥ وَمَا هُمُۥ بِحَٰمِلِينَ مِنۡ خَطَٰيَٰهُمُۥ مِن شَيۡءٍۖ إِنَّهُمُۥ لَكَٰذِبُونَ _ وَلَيَحۡمِلُنَّ أَثۡقَالَهُمُۥ وَأَثۡقَالٗا مَّعَ أَثۡقَالِهِمُۥۖ وَلَيُسۡـَٔلُنَّ يَوۡمَ ٱلۡقِيَٰمَةِ عَمَّا كَانُواْ يَفۡتَرُونَ _ وَلَقَدۡ أَرۡسَلۡنَا نُوحًا إِلَىٰ قَوۡمِهِۦ فَلَبِثَ فِيهِمُۥ أَلۡفَ سَنَةٍ إِلَّا خَمۡسِينَ عَامٗا فَأَخَذَهُمُ ٱلطُّوفَانُ وَهُمُۥ ظَٰلِمُونَ _ فَأَنجَيۡنَٰهُۥ وَأَصۡحَٰبَ ٱلسَّفِينَةِ وَجَعَلۡنَٰهَا ءَايَةٗ لِّلۡعَٰلَمِينَ _ وَإِبۡرَٰهِيمَ إِذۡ قَالَ لِقَوۡمِهِ ٱعۡبُدُواْ ٱللَّهَ وَٱتَّقُوهُۥۖ ذَٰلِكُمُۥ خَيۡرٞ لَّكُمُۥ إِن كُنتُمُۥ تَعۡلَمُونَ _ إِنَّمَا تَعۡبُدُونَ مِن دُونِ ٱللَّهِ أَوۡثَٰنٗا وَتَخۡلُقُونَ إِفۡكًاۚ إِنَّ ٱلَّذِينَ تَعۡبُدُونَ مِن دُونِ ٱللَّهِ لَا يَمۡلِكُونَ لَكُمُۥ رِزۡقٗا فَٱبۡتَغُواْ عِندَ ٱللَّهِ ٱلرِّزۡقَ وَٱعۡبُدُوهُۥ وَٱشۡكُرُواْ لَهُۥۖ إِلَيۡهِۦ تُرۡجَعُونَ _ وَإِن تُكَذِّبُواْ فَقَدۡ كَذَّبَ أُمَمٞ مِّن قَبۡلِكُمُۥۖ وَمَا عَلَى ٱلرَّسُولِ إِلَّا ٱلۡبَلَٰغُ ٱلۡمُبِينُ _ أَوَلَمۡ يَرَوۡاْ كَيۡفَ يُبۡدِئُ ٱللَّهُ ٱلۡخَلۡقَ ثُمَّ يُعِيدُهُۥۚ إِنَّ ذَٰلِكَ عَلَى ٱللَّهِ يَسِيرٞ _ قُلۡ سِيرُواْ فِي ٱلۡأَرۡضِ فَٱنظُرُواْ كَيۡفَ بَدَأَ ٱلۡخَلۡقَۚ ثُمَّ ٱللَّهُ يُنشِئُ ٱلنَّشَآءَةَ ٱلۡأٓخِرَةَۚ إِنَّ ٱللَّهَ عَلَىٰ كُلِّ شَيۡءٖ قَدِيرٞ _ يُعَذِّبُ مَن يَشَآءُ وَيَرۡحَمُ مَن يَشَآءُۖ وَإِلَيۡهِۦ تُقۡلَبُونَ _ وَمَا أَنتُمُۥ بِمُعۡجِزِينَ فِي ٱلۡأَرۡضِ وَلَا فِي ٱلسَّمَآءِۖ وَمَا لَكُمُۥ مِن دُونِ ٱللَّهِ مِن وَلِيّٖ وَلَا نَصِيرٖ _ وَٱلَّذِينَ كَفَرُواْ بِـَٔايَٰتِ ٱللَّهِ وَلِقَآئِهِۦ أُوْلَٰٓئِكَ يَئِسُواْ مِن رَّحۡمَتِي وَأُوْلَٰٓئِكَ لَهُمُۥ عَذَابٌ أَلِيمٞ _ فَمَا كَانَ جَوَابَ قَوۡمِهِۦ إِلَّا أَن قَالُواْ ٱقۡتُلُوهُۥ أَوۡ حَرِّقُوهُۥ فَأَنجَىٰهُ ٱللَّهُ مِنَ ٱلنَّارِۚ إِنَّ فِي ذَٰلِكَ لَأٓيَٰتٖ لِّقَوۡمٖ يُؤۡمِنُونَ _ وَقَالَ إِنَّمَا ٱتَّخَذۡتُمُۥ مِن دُونِ ٱللَّهِ أَوۡثَٰنٗا مَّوَدَّةُ بَيۡنِكُمُۥ فِي ٱلۡحَيَوٰةِ ٱلدُّنۡيَاۖ ثُمَّ يَوۡمَ ٱلۡقِيَٰمَةِ يَكۡفُرُ بَعۡضُكُمُۥ بِبَعۡضٖ وَيَلۡعَنُ بَعۡضُكُمُۥ بَعۡضٗا وَمَأۡوَىٰكُمُ ٱلنَّارُ وَمَا لَكُمُۥ مِن نَّٰصِرِينَ _ ۞فَـَٔامَنَ لَهُۥ لُوطٞۘ وَقَالَ إِنِّي مُهَاجِرٌ إِلَىٰ رَبِّيۖ إِنَّهُۥ هُوَ ٱلۡعَزِيزُ ٱلۡحَكِيمُ _ وَوَهَبۡنَا لَهُۥ إِسۡحَٰقَ وَيَعۡقُوبَ وَجَعَلۡنَا فِي ذُرِّيَّتِهِ ٱلنُّبُوَّةَ وَٱلۡكِتَٰبَ وَءَاتَيۡنَٰهُۥ أَجۡرَهُۥ فِي ٱلدُّنۡيَاۖ وَإِنَّهُۥ فِي ٱلۡأٓخِرَةِ لَمِنَ ٱلصَّٰلِحِينَ _ وَلُوطًا إِذۡ قَالَ لِقَوۡمِهِۦ إِنَّكُمُۥ لَتَأۡتُونَ ٱلۡفَٰحِشَةَ مَا سَبَقَكُمُۥ بِهَا مِنۡ أَحَدٖ مِّنَ ٱلۡعَٰلَمِينَ _ أَى۪نَّكُمُۥ لَتَأۡتُونَ ٱلرِّجَالَ وَتَقۡطَعُونَ ٱلسَّبِيلَ 28 وَتَأۡتُونَ فِي نَادِيكُمُ ٱلۡمُنكَرَۖ فَمَا كَانَ جَوَابَ قَوۡمِهِۦ إِلَّا أَن قَالُواْ ٱئۡتِنَا بِعَذَابِ ٱللَّهِ إِن كُنتَ مِنَ ٱلصَّٰدِقِينَ _ قَالَ رَبِّ ٱنصُرۡنِي عَلَى ٱلۡقَوۡمِ ٱلۡمُفۡسِدِينَ _ وَلَمَّا جَآءَتۡ رُسُلُنَا إِبۡرَٰهِيمَ بِٱلۡبُشۡرَىٰ قَالُواْ إِنَّا مُهۡلِكُواْ أَهۡلِ هَٰذِهِ ٱلۡقَرۡيَةِۖ إِنَّ أَهۡلَهَا كَانُواْ ظَٰلِمِينَ _ قَالَ إِنَّ فِيهَا لُوطٗاۚ قَالُواْ نَحۡنُ أَعۡلَمُ بِمَن فِيهَاۖ لَنُنَجِّيَنَّهُۥ وَأَهۡلَهُۥ إِلَّا ٱمۡرَأَتَهُۥ كَانَتۡ مِنَ ٱلۡغَٰبِرِينَ _ وَلَمَّا أَن جَآءَتۡ رُسُلُنَا لُوطٗا سِيٓءَ بِهِمُۥ وَضَاقَ بِهِمُۥ ذَرۡعٗاۖ وَقَالُواْ لَا تَخَفۡ وَلَا تَحۡزَنۡ إِنَّا مُنجُوكَ وَأَهۡلَكَ إِلَّا ٱمۡرَأَتَكَ كَانَتۡ مِنَ ٱلۡغَٰبِرِينَ _ إِنَّا مُنزِلُونَ عَلَىٰ أَهۡلِ هَٰذِهِ ٱلۡقَرۡيَةِ رِجۡزٗا مِّنَ ٱلسَّمَآءِ بِمَا كَانُواْ يَفۡسُقُونَ _ وَلَقَد تَّرَكۡنَا مِنۡهَا ءَايَةَۢ بَيِّنَةٗ لِّقَوۡمٖ يَعۡقِلُونَ _ وَإِلَىٰ مَدۡيَنَ أَخَاهُمُۥ شُعَيۡبٗا فَقَالَ يَٰقَوۡمِ ٱعۡبُدُواْ ٱللَّهَ وَٱرۡجُواْ ٱلۡيَوۡمَ ٱلۡأٓخِرَ وَلَا تَعۡثَوۡاْ فِي ٱلۡأَرۡضِ مُفۡسِدِينَ _ فَكَذَّبُوهُۥ فَأَخَذَتۡهُمُ ٱلرَّجۡفَةُ فَأَصۡبَحُواْ فِي دَارِهِمُۥ جَٰثِمِينَ _ وَعَادٗا وَثَمُودٗا وَقَد تَّبَيَّنَ لَكُمُۥ مِن مَّسَٰكِنِهِمُۥۖ وَزَيَّنَ لَهُمُ ٱلشَّيۡطَٰنُ أَعۡمَٰلَهُمُۥ فَصَدَّهُمُۥ عَنِ ٱلسَّبِيلِ وَكَانُواْ مُسۡتَبۡصِرِينَ _ وَقَٰرُونَ وَفِرۡعَوۡنَ وَهَٰمَٰنَۖ وَلَقَدۡ جَآءَهُمُۥ مُوسَىٰ بِٱلۡبَيِّنَٰتِ فَٱسۡتَكۡبَرُواْ فِي ٱلۡأَرۡضِ وَمَا كَانُواْ سَٰبِقِينَ _ فَكُلًّا أَخَذۡنَا بِذَنۢبِهِۦۖ فَمِنۡهُمُۥ مَنۡ أَرۡسَلۡنَا عَلَيۡهِۦ حَاصِبٗا وَمِنۡهُمُۥ مَنۡ أَخَذَتۡهُ ٱلصَّيۡحَةُ وَمِنۡهُمُۥ مَنۡ خَسَفۡنَا بِهِ ٱلۡأَرۡضَ وَمِنۡهُمُۥ مَنۡ أَغۡرَقۡنَاۚ وَمَا كَانَ ٱللَّهُ لِيَظۡلِمَهُمُۥ وَلَٰكِن كَانُواْ أَنفُسَهُمُۥ يَظۡلِمُونَ _ مَثَلُ ٱلَّذِينَ ٱتَّخَذُواْ مِن دُونِ ٱللَّهِ أَوۡلِيَآءَ كَمَثَلِ ٱلۡعَنكَبُوتِ ٱتَّخَذَتۡ بَيۡتٗاۖ وَإِنَّ أَوۡهَنَ ٱلۡبِيُوتِ لَبَيۡتُ ٱلۡعَنكَبُوتِۚ لَوۡ كَانُواْ يَعۡلَمُونَ _ إِنَّ ٱللَّهَ يَعۡلَمُ مَا تَدۡعُونَ مِن دُونِهِۦ مِن شَيۡءٖۚ وَهُوَ ٱلۡعَزِيزُ ٱلۡحَكِيمُ _ وَتِلۡكَ ٱلۡأَمۡثَٰلُ نَضۡرِبُهَا لِلنَّاسِۖ وَمَا يَعۡقِلُهَا إِلَّا ٱلۡعَٰلِمُونَ _ خَلَقَ ٱللَّهُ ٱلسَّمَٰوَٰتِ وَٱلۡأَرۡضَ بِٱلۡحَقِّۚ إِنَّ فِي ذَٰلِكَ لَأٓيَةٗ لِّلۡمُؤۡمِنِينَ _ ٱتۡلُ مَا أُوحِيَ إِلَيۡكَ مِنَ ٱلۡكِتَٰبِ وَأَقِمِ ٱلصَّلَوٰةَۖ إِنَّ ٱلصَّلَوٰةَ تَنۡهَىٰ عَنِ ٱلۡفَحۡشَآءِ وَٱلۡمُنكَرِۗ وَلَذِكۡرُ ٱللَّهِ أَكۡبَرُۗ وَٱللَّهُ يَعۡلَمُ مَا تَصۡنَعُونَ _ ۞وَلَا تُجَٰدِلُواْ أَهۡلَ ٱلۡكِتَٰبِ إِلَّا بِٱلَّتِي هِيَ أَحۡسَنُ إِلَّا ٱلَّذِينَ ظَلَمُواْ مِنۡهُمُۥۖ وَقُولُواْ ءَامَنَّا بِٱلَّذِي أُنزِلَ إِلَيۡنَا وَأُنزِلَ إِلَيۡكُمُۥ وَإِلَٰهُنَا وَإِلَٰهُكُمُۥ وَٰحِدٞ وَنَحۡنُ لَهُۥ مُسۡلِمُونَ _ وَكَذَٰلِكَ أَنزَلۡنَا إِلَيۡكَ ٱلۡكِتَٰبَۚ فَٱلَّذِينَ ءَاتَيۡنَٰهُمُ ٱلۡكِتَٰبَ يُؤۡمِنُونَ بِهِۦۖ وَمِنۡ هَٰؤُلَآءِ مَن يُؤۡمِنُ بِهِۦۚ وَمَا يَجۡحَدُ بِـَٔايَٰتِنَا إِلَّا ٱلۡكَٰفِرُونَ _ وَمَا كُنتَ تَتۡلُواْ مِن قَبۡلِهِۦ مِن كِتَٰبٖ وَلَا تَخُطُّهُۥ بِيَمِينِكَۖ إِذٗا لَّٱرۡتَابَ ٱلۡمُبۡطِلُونَ _ بَلۡ هُوَ ءَايَٰتُۢ بَيِّنَٰتٞ فِي صُدُورِ ٱلَّذِينَ أُوتُواْ ٱلۡعِلۡمَۚ وَمَا يَجۡحَدُ بِـَٔايَٰتِنَا إِلَّا ٱلظَّٰلِمُونَ _ وَقَالُواْ لَوۡلَا أُنزِلَ عَلَيۡهِۦ ءَايَتٞ مِّن رَّبِّهِۦۚ قُلۡ إِنَّمَا ٱلۡأٓيَٰتُ عِندَ ٱللَّهِ وَإِنَّمَا أَنَا۠ نَذِيرٞ مُّبِينٌ _ أَوَلَمۡ يَكۡفِهِمُۥ أَنَّا أَنزَلۡنَا عَلَيۡكَ ٱلۡكِتَٰبَ يُتۡلَىٰ عَلَيۡهِمُۥۚ إِنَّ فِي ذَٰلِكَ لَرَحۡمَةٗ وَذِكۡرَىٰ لِقَوۡمٖ يُؤۡمِنُونَ _ قُلۡ كَفَىٰ بِٱللَّهِ بَيۡنِي وَبَيۡنَكُمُۥ شَهِيدٗاۖ يَعۡلَمُ مَا فِي ٱلسَّمَٰوَٰتِ وَٱلۡأَرۡضِۗ وَٱلَّذِينَ ءَامَنُواْ بِٱلۡبَٰطِلِ وَكَفَرُواْ بِٱللَّهِ أُوْلَٰٓئِكَ هُمُ ٱلۡخَٰسِرُونَ _ وَيَسۡتَعۡجِلُونَكَ بِٱلۡعَذَابِ وَلَوۡلَا أَجَلٞ مُّسَمّٗى لَّجَآءَهُمُ ٱلۡعَذَابُۚ وَلَيَأۡتِيَنَّهُمُۥ بَغۡتَةٗ وَهُمُۥ لَا يَشۡعُرُونَ _ يَسۡتَعۡجِلُونَكَ بِٱلۡعَذَابِ وَإِنَّ جَهَنَّمَ لَمُحِيطَةُۢ بِٱلۡكَٰفِرِينَ _ يَوۡمَ يَغۡشَىٰهُمُ ٱلۡعَذَابُ مِن فَوۡقِهِمُۥ وَمِن تَحۡتِ أَرۡجُلِهِمُۥ وَنَقُولُ ذُوقُواْ مَا كُنتُمُۥ تَعۡمَلُونَ _ يَٰعِبَادِيَ ٱلَّذِينَ ءَامَنُواْ إِنَّ أَرۡضِي وَٰسِعَةٞ فَإِيَّٰيَ فَٱعۡبُدُونِ _ كُلُّ نَفۡسٖ ذَآئِقَةُ ٱلۡمَوۡتِۖ ثُمَّ إِلَيۡنَا تُرۡجَعُونَ _ وَٱلَّذِينَ ءَامَنُواْ وَعَمِلُواْ ٱلصَّٰلِحَٰتِ لَنُبَوِّئَنَّهُمُۥ مِنَ ٱلۡجَنَّةِ غُرَفٗا تَجۡرِي مِن تَحۡتِهَا ٱلۡأَنۡهَٰرُ خَٰلِدِينَ فِيهَاۚ نِعۡمَ أَجۡرُ ٱلۡعَٰمِلِينَ _ ٱلَّذِينَ صَبَرُواْ وَعَلَىٰ رَبِّهِمُۥ يَتَوَكَّلُونَ _ وَكَآئِن مِّن دَآبَّةٖ لَّا تَحۡمِلُ رِزۡقَهَا ٱللَّهُ يَرۡزُقُهَا وَإِيَّاكُمُۥۚ وَهُوَ ٱلسَّمِيعُ ٱلۡعَلِيمُ _ وَلَئِن سَأَلۡتَهُمُۥ مَنۡ خَلَقَ ٱلسَّمَٰوَٰتِ وَٱلۡأَرۡضَ وَسَخَّرَ ٱلشَّمۡسَ وَٱلۡقَمَرَ لَيَقُولُنَّ ٱللَّهُۖ فَأَنَّىٰ يُؤۡفَكُونَ _ ٱللَّهُ يَبۡسُطُ ٱلرِّزۡقَ لِمَن يَشَآءُ مِنۡ عِبَادِهِۦ وَيَقۡدِرُ لَهُۥۚ إِنَّ ٱللَّهَ بِكُلِّ شَيۡءٍ عَلِيمٞ _ وَلَئِن سَأَلۡتَهُمُۥ مَن نَّزَّلَ مِنَ ٱلسَّمَآءِ مَآءٗ فَأَحۡيَا بِهِ ٱلۡأَرۡضَ مِنۢ بَعۡدِ مَوۡتِهَا لَيَقُولُنَّ ٱللَّهُۚ قُلِ ٱلۡحَمۡدُ لِلَّهِۚ بَلۡ أَكۡثَرُهُمُۥ لَا يَعۡقِلُونَ _ وَمَا هَٰذِهِ ٱلۡحَيَوٰةُ ٱلدُّنۡيَا إِلَّا لَهۡوٞ وَلَعِبٞۚ وَإِنَّ ٱلدَّارَ ٱلۡأٓخِرَةَ لَهِيَ ٱلۡحَيَوَانُۚ لَوۡ كَانُواْ يَعۡلَمُونَ _ فَإِذَا رَكِبُواْ فِي ٱلۡفُلۡكِ دَعَوُاْ ٱللَّهَ مُخۡلِصِينَ لَهُ ٱلدِّينَ فَلَمَّا نَجَّىٰهُمُۥ إِلَى ٱلۡبَرِّ إِذَا هُمُۥ يُشۡرِكُونَ _ لِيَكۡفُرُواْ بِمَا ءَاتَيۡنَٰهُمُۥ وَلۡيَتَمَتَّعُواْۚ فَسَوۡفَ يَعۡلَمُونَ _ أَوَلَمۡ يَرَوۡاْ أَنَّا جَعَلۡنَا حَرَمًا ءَامِنٗا وَيُتَخَطَّفُ ٱلنَّاسُ مِنۡ حَوۡلِهِمُۥۚ أَفَبِٱلۡبَٰطِلِ يُؤۡمِنُونَ وَبِنِعۡمَةِ ٱللَّهِ يَكۡفُرُونَ _ وَمَنۡ أَظۡلَمُ مِمَّنِ ٱفۡتَرَىٰ عَلَى ٱللَّهِ كَذِبًا أَوۡ كَذَّبَ بِٱلۡحَقِّ لَمَّا جَآءَهُۥۚ أَلَيۡسَ فِي جَهَنَّمَ مَثۡوٗى لِّلۡكَٰفِرِينَ _ وَٱلَّذِينَ جَٰهَدُواْ فِينَا لَنَهۡدِيَنَّهُمُۥ سُبُلَنَاۚ وَإِنَّ ٱللَّهَ لَمَعَ ٱلۡمُحۡسِنِينَ _
